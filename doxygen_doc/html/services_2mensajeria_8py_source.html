94 )

147def _room_name(a, b):
148 """
149 Construye un nombre de sala único para dos usuarios,
150 forzando ambos IDs a entero para evitar TypeError.
151 """
152 try:
153 a_int = int(a)
154 b_int = int(b)
155 except (TypeError, ValueError):
156 return None
157
158 a_int, b_int = sorted([a_int, b_int])
159 return f"chat_{a_int}_{b_int}"
